### 3.1.3 GetStart

```jsx
import { useState } from 'react';
import { DatePicker, message, Alert } from 'antd';

export default function GetStart() {
  const [date, setDate] = useState(null);
  const handleChange = (value) => {
    message.info(`Selected Date: ${value ? value.format('YYYY-MM-DD') : 'None'}`);
    setDate(value);
  };
  return (
    <div style={{ width: 400, margin: '100px auto' }}>
      <DatePicker onChange={handleChange} />
      <div style={{ marginTop: 16 }}>
        <Alert message='Selected Date' description={date ? date.format('YYYY-MM-DD') : 'None'} />
      </div>
    </div>
  );
}
```

```jsx
import React, { useState } from 'react';
import { Alert, Button } from 'react-bootstrap';

export default function AlertDismissible() {
  const [show, setShow] = useState(false);

  return (
    <>
      <Alert show={show} variant='success'>
        <Alert.Heading>How's it going?!</Alert.Heading>
        <p>
          Duis mollis, est non commodo luctus, nisi erat porttitor ligula, eget lacinia odio sem nec elit. Cras mattis consectetur purus sit amet fermentum.
        </p>
        <hr />
        <div className='d-flex justify-content-end'>
          <Button onClick={() => setShow(false)} variant='outline-success'>
            Close me y'all!
          </Button>
        </div>
      </Alert>

      {!show && <Button onClick={() => setShow(true)}>Show Alert</Button>}
    </>
  );
}
```

# 7. Tailwindcss

https://tailwindcss.com/docs/installation

- Utility-first CSS framework,
- 즉 미리 정의된 CSS의 특성들을 `class` 를 통해 불러와 적용시키는 형태의 CSS 프레임워크
- 원하는 `html` 요소에 `class` 을 부여해주면 바로 사용이 가능하다

cf. VSCode 확장기능

- Tailwind CSS IntelliSense : 클래스명 자동완성
- Tailwind Docs : 문서 찾아볼 떄 Ctrl + Shipt + P로 검색

---

## 7.1 예제 1 : 기본 사용법

https://tailwindcss.com/docs/installation

```sh
npm install -D tailwindcss
npx tailwindcss init
```

tailwindcss를 사용하고자 하는 경로, 파일 작성(**확장자명은 띄워쓰면 안됨**)

```js
/** @type {import('tailwindcss').Config} */
module.exports = {
  content: ['./src/**/*.{html,js,jsx,ts,tsx}'],
  theme: {
    extend: {},
  },
  plugins: [],
};
```

전역 CSS에 다음 코드 추가 (CRA에서는 index.css 맨 위에 추가)

```css
@tailwind base; /* tailwind 기본 스타일 및 확장된 스타일 주입 */
@tailwind components; /* tailwind 기본 컴포넌트 클래스 및 확장된 컴포넌트 클래스 주입 */
@tailwind utilities; /* tailwind 기본 유틸리티 클래스 및 확장된 유틸리티 클래스 주입 */
```

# 8. 정리

- 필요에 따라서 라이브러리를 찾고(alternatives)
- 공식 문서로 라이브러리에 대해 파악하고(Introduction)
- 공식 문서 따라 설치하고(Installation)
- 공식 문서 예제(Examples) 따라해보고 api 문서 보고 적용해본다

라이브러리의 트렌드 또한 빠르게 변한다.
2년전에 당연했던 선택들이 지금은 당연하지 않다.
결국 스스로 새로운 라이브러리를 적용해보는 능력(경험)이 관건!

라이브러리의 적용은 정답이 없다. 맞는 것도 없지만 틀린 것도 없다.
만든 이의 설명과 가이드에 따라 내 프로젝트에 적용하는 것으로 성공적임
계속 시도 / 반복 시도 / 새로운 기술에 대한 호기심과 궁금증으로 Just do it!
